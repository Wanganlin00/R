2  数据类型和数据结构

3 数据类型

数据类型（mode）：表示对象在计算机内存中的存储形式

3.1 numeric – 数值型（integer/double）

integer – 整型 L

mode(4.3)
#> [1] "numeric"
class(4.3)
#> [1] "numeric"
mode(1L)
#> [1] "numeric"
class(1L)
#> [1] "integer"

3.2 character – 字符型

mode(c("car"))
#> [1] "character"
class("car")
#> [1] "character"

3.3 logical – 逻辑型

mode(c(TRUE,FALSE))
#> [1] "logical"
class(c(TRUE,FALSE))
#> [1] "logical"

3.4 complex – 复数型


mode(c(1+2i,3-4i)) 
#> [1] "complex"
class(c(1+2i,3-4i))
#> [1] "complex"

3.5 raw型

charToRaw("abcde12345") #字符串中每个字符的原始存储格式（十六进制？）
#>  [1] 61 62 63 64 65 31 32 33 34 35
mode(charToRaw("abcde12345")) 
#> [1] "raw"
class(charToRaw("abcde12345"))
#> [1] "raw"

4 数据结构

class：是一种基于面向对象的R的抽象类型划分,或者理解为一种数据结构。

4.1 vector

向量,同一类元素的集合。 ?c #Combine Values into a Vector or List

R中没有标量（scalar），只含一个元素的向量

#单元素向量
1 == c(1)
#> [1] TRUE
is.vector(1)
#> [1] TRUE
  
"a"
#> [1] "a"
is.vector("a")
#> [1] TRUE

#使用函数c()创建向量

c(1)
#> [1] 1
c(1,2,3,4,5)
#> [1] 1 2 3 4 5
c("a","b","c")
#> [1] "a" "b" "c"

4.2 factor

因子，分类变量，其中每个级别都是一类。?factor factor(vector,ordered=FALSE,levels=c(v1,v2,…),labels= ,...) ，在内存中以整数向量c(1,2,3,...,k)存储。

# 存储形式
class(factor((c(1,2,3)))) 
#> [1] "factor"
mode(factor((c(1,2,3)))) 
#> [1] "numeric"

# 名义变量 nominal variable
diabetes<-c("t1","t2","t1","t1") 
str(diabetes)
#>  chr [1:4] "t1" "t2" "t1" "t1"
diabetes<-factor(diabetes)
str(diabetes)
#>  Factor w/ 2 levels "t1","t2": 1 2 1 1

# 顺序变量 ordinal variable     默认水平根据字母顺序而定
status<-c("poor","better","best","poor")
status<-factor(status,ordered = TRUE) 
str(status) 
#>  Ord.factor w/ 3 levels "best"<"better"<..: 3 2 1 3
status<-factor(status,ordered =TRUE,levels = c("poor","better","best")) 
str(status) 
#>  Ord.factor w/ 3 levels "poor"<"better"<..: 1 2 3 1


#改变外在标签
sex<-c(1,2,2,1)
sex
#> [1] 1 2 2 1
sex<-factor(sex,levels=c(1,2),labels = c("男","女")) 
str(sex) 
#>  Factor w/ 2 levels "男","女": 1 2 2 1
sex
#> [1] 男 女 女 男
#> Levels: 男 女

leadership <- read_csv("data/leadership.txt") |> column_to_rownames(var = "id")
#> Rows: 5 Columns: 10
#> ── Column specification ────────────────────────────────────────────────────────
#> Delimiter: ","
#> chr (4): id, date, country, gender
#> dbl (6): age, q1, q2, q3, q4, q5
#> 
#> ℹ Use `spec()` to retrieve the full column specification for this data.
#> ℹ Specify the column types or set `show_col_types = FALSE` to quiet this message.
# 连续型变量→因子
leadership$agecat<-cut(leadership$age,
                    breaks = c(0,18,45,65,Inf),
                    labels=c("minor","young","middle_age","elder"),
                    include.lowest = TRUE,
                    right = TRUE
                    )
leadership$agecat
#> [1] young young young elder elder
#> Levels: minor young middle_age elder

class(as.Date("2023-12-11"))

format(Sys.Date(),"%Y/%m/%d") #以"%Y/%m/%d"格式输出

4.4 matrix

矩阵，所有元素必须是同一类型。?matrix

matrix(data= ,nrow=1 ,ncol=1 ,byrow=FALSE ,dimnames=list(rnames,cnames) ,...)

num<-c(16,22,24,28)
rnames<-c("R1","R2")
cnames<-c("C1","C2")
mymatrix<-matrix(num,nrow=2,ncol=2,byrow=TRUE,dimnames=list(rnames,cnames))
mymatrix   
#>    C1 C2
#> R1 16 22
#> R2 24 28

4.5 array

数组，所有元素必须是同一类型。?array

array(data,dim_numeric_vector=c(...),dimnames = list(dim1,dim2,...),...)

v<-1:24 
dim1<-c("A1","A2","A3") 
dim2<-c("B1","B2","B3","B4")
dim3<-c("C1","C2") 
myarray<-array(v,c(3,4,2),dimnames = list(dim1,dim2,dim3)) 
myarray
#> , , C1
#> 
#>    B1 B2 B3 B4
#> A1  1  4  7 10
#> A2  2  5  8 11
#> A3  3  6  9 12
#> 
#> , , C2
#> 
#>    B1 B2 B3 B4
#> A1 13 16 19 22
#> A2 14 17 20 23
#> A3 15 18 21 24

4.6 data.frame/tibble

数据框， 由列向量组成，每一列元素必须是同一类型，列与列之间长度必须相同，但类型可以不同。?data.frame

data.frame(name1=col_vector1,name2=col_vector2,...,row.names = ,...) ?tibble，tibble()是tidyverse风格的数据框，用法类似。

id<-c(1,2,3,4) 
age<-c(21,14,52,15) 
diabetes<-c("t1","t2","t1","t1") 
status<-c("poor","better","best","poor") 
patient<-data.frame(patientID=id,age,diabetes,status,row.names = id) # 4个列向量组成数据框 
patient |> rownames_to_column(var = "ID")
#>   ID patientID age diabetes status
#> 1  1         1  21       t1   poor
#> 2  2         2  14       t2 better
#> 3  3         3  52       t1   best
#> 4  4         4  15       t1   poor
mode(patient)
#> [1] "list"
class(patient)
#> [1] "data.frame"

4.7 list

列表，由以上各种数据结构组成?list list(name1=object1,name2=object2,...)

mylist<-list(title="My list",
             matr=matrix(c("a1","b1","a2","b2"),nrow=2,ncol=2,byrow=TRUE,
                           dimnames = list(c("X1","X2"),c("Y1","Y2"))
                         ),
             df=data.frame(id=matrix(c("Lisa","BOb","John","Jule"),
                                     nrow=4,ncol=1,byrow=TRUE
                                     ),
                           int=c(3,5,7,9),
                           TF=c(T,T,T,F)
                           ),
             list=list(a=c(1,2,3),b=c("A","B"))
             )
mylist

#> [1] "My list"
#> 
#> $matr

#> X1 "a1" "b1"
#> X2 "a2" "b2"
#> 
#> $df
#>     id int    TF
#> 1 Lisa   3  TRUE
#> 2  BOb   5  TRUE
#> 3 John   7  TRUE
#> 4 Jule   9 FALSE
#> 
#> $list
#> $list$a
#> [1] 1 2 3
#> 
#> $list$b
#> [1] "A" "B"
